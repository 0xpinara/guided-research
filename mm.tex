\section{Feature Engineering Performed}

\subsection{Feature Families Computed}
The pipeline computes:
\begin{itemize}[leftmargin=1.2em]
  \item \textbf{Options features} \texttt{feat\_01}--\texttt{feat\_26}: put/call ratios, IV level/rank/skew/term slope, 0DTE share, spread/moneyness, GEX, charm exposure, BKM-style implied skewness and kurtosis proxy.
  \item \textbf{Stock + macro controls} \texttt{feat\_27}--\texttt{feat\_42}: returns, HV, RSI, MACD, Bollinger position, ATR, VIX controls, IRX, sector ETF return.
  \item \textbf{Event/temporal features} \texttt{feat\_43}--\texttt{feat\_48}: days to earnings/ex-dividend, earnings-within-7d flag, day-of-week, days to OpEx, quarter-end flag.
\end{itemize}

\subsection{Target Variables}
\begin{itemize}[leftmargin=1.2em]
  \item \texttt{target\_r\_t\_plus\_1} $= \frac{P_{t+1}}{P_t} - 1$
  \item \texttt{target\_direction\_t\_plus\_1} $= \mathbb{1}(target\_r\_t\_plus\_1 > 0)$
\end{itemize}

\section{Model Training Suite Implemented}
Training script: \texttt{train\_panel\_model\_suite.py}.

\subsection{Architectures}
\begin{itemize}[leftmargin=1.2em]
  \item \textbf{FFNN}: $256 \rightarrow 128 \rightarrow 64 \rightarrow 32$ with ReLU + Dropout.
  \item \textbf{XGBoost}: regressor and classifier, tree-based baseline.
  \item \textbf{TabNet}: regressor + classifier with attention masks.
  \item \textbf{TFT}: one-step horizon, 20-day encoder window, static categorical (\texttt{ticker}, \texttt{sector}), known/unknown time-varying inputs.
\end{itemize}

\subsection{Feature Set Design (A/B/C)}
\begin{itemize}[leftmargin=1.2em]
  \item Model A: stock-only (\texttt{feat\_27}--\texttt{feat\_42}, 16 features)
  \item Model B: options-only (\texttt{feat\_01}--\texttt{feat\_26}, 26 features)
  \item Model C: all (\texttt{feat\_01}--\texttt{feat\_48}, 48 features)
\end{itemize}

\subsection{Benchmarks}
\begin{itemize}[leftmargin=1.2em]
  \item naive zero,
  \item historical mean,
  \item OLS full,
  \item OLS on BS-related subset,
  \item single-feature linear predictors (put-call ratio, IV change, parity-proxy).
\end{itemize}
